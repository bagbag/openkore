automacro warpmap {
	timeout 10

	call {
		$lckMap = @config(lockMap);
		
		if ($.map != $lckMap) {
			do c @warp $lckMap
		}
	}
}

automacro autoaloot {
	delay 15
	timeout 300
	call aloot
}

macro berrybox {
	$autorestart = @config (autoRestart)
	log autorestart was $autorestart
	do conf autoRestart 0

	do ai manual
	do c @die
	do c @warp prt_in 35 98
	do look 2 0
	
	$berrynpc = @npc ("Berry Exchanger")
	$exit = 0
	
	while ($exit == 0) as berryboxloop
	
		log inventory: @invamount(Yggdrasil Berry) berrys left
		while (@invamount(Yggdrasil Berry) >= 10) as npcloop
			do talk $berrynpc
			do talk resp 0
		end npcloop
		
		do c @storage
		pause 1
		log storage: @storamount(Yggdrasil Berry) berrys left
		if (@storamount(Yggdrasil Berry) >= 10) {
			do storage get Yggdrasil Berry 100
		} else { 
			$exit = 1
			log berrybox done
		}
		do storage close
	end berryboxloop
	
	do conf autoRestart autorestart
	do ai auto
}
